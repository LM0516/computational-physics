\section{Numerical Integration}

\subsection{Newton-Cotes Integration}

\begin{exercise}\label{ex:integrals}
	For each integral below, use the trapezoidal rule to estimate the integral
	for $m=10\cdot 2^k$ nodes for $k=1,2,\ldots,10$. Make a log-log plot of the
	errors and confirm or refute second-order accuracy.
	\begin{enumerate}
		\item $\displaystyle \int_0^1 x\log(1+x)\, dx = \frac{1}{4}$

		\item $\displaystyle \int_0^1 x^2 \tan^{-1}x\, dx = \frac{\pi-2+2\log 2}{12}$

		\item $\displaystyle \int_0^{\pi/2}e^x \cos x\, dx = \frac{e^{\pi/2}-1}{2}$

		\item $\displaystyle \int_0^1 {\tan^{-1}(\sqrt{2+x^2})\over
				      (1+x^2)\sqrt{2+x^2}}dx={5\pi^2\over 96}$

		\item $\displaystyle \int_0^1 \sqrt{x} \log(x) \, dx = -\frac{4}{9}$
		      (Hint: Although the integrand has the limiting value zero as $x\to 0$,
		      it cannot be evaluated naively at $x=0$. One possibility is to start the
		      integral at $x=\epsilon_{\rm mach}$. Alternatively, you can define a
		      wrapper function $g(x)$ that returns $g(0)=0$ and $g(x)=\sqrt{x}\log(x)$
		      otherwise.)

		\item $\displaystyle \int_0^1 \sqrt{1-x^2}\,\, dx = \frac{\pi}{4}$
	\end{enumerate}
\end{exercise}

\begin{solution}
	To evaluate the given definite integrals, we employ the composite
	trapezoidal rule. For an interval $[a, b]$ discretized into $m$ equally
	spaced sub-intervals with step size $h = \frac{b-a}{m}$, the composite
	trapezoidal rule approximations are given by:
	\[
		\int_a^b f(x) \, dx \approx \frac{h}{2} \left( f(x_0) +
		2\sum_{i=1}^{m-1} f(x_i) + f(x_m) \right)
	\]
	where $x_i = a + i h$. The number of nodes was systematically increased
	according to $m = 10 \cdot 2^k$ for $k = 1, 2, \ldots, 10$, yielding a
	maximum of $10,240$ nodes.

	\paragraph{Results}
	Table~\ref{tab:integral_results} summarizes the final numerical

	theoretical exact values.

	\paragraph{Discussion \& Convergence Analysis}
	The error bound for the composite trapezoidal rule is theoretically given by:
	\[
		E \leq \frac{(b-a)}{12} h^2 \max_{x \in [a,b]} |f''(x)| = \mathcal{O}(h^2)
	\]
	This stipulates that the algorithm demonstrates second-order accuracy
	\textit{only} if the integrand $f(x)$ is twice continuously differentiable,
	$f \in C^2[a,b]$.

	As observed in Table~\ref{tab:integral_results} and
	Figure~\ref{fig:abs_error}, all the integrals align with theoretical
	expectations, exhibiting strict second-order accuracy. In a log-log plot of
	error versus nodes $m$, this manifests as a line with a slope of $-2$.

\end{solution}

\begin{exercise}
	For each integral in Exercise~\ref{ex:integrals} above, apply the Simpson
	formula and compare the errors to fourth-order convergence.
\end{exercise}

\begin{solution}
	To evaluate the convergence improvements over the trapezoidal rule, we apply
	the composite Simpson's rule to the same set of integrals. For an interval
	$[a, b]$ discretized into $m$ equally spaced sub-intervals (where $m$ is even)
	with step size $h = \frac{b-a}{m}$, the approximation is given by:
	\[
		\int_a^b f(x) \, dx \approx \frac{h}{3} \left[ f(x_0) + 4
			\sum_{i=1,3,5\ldots}^{m-1} f(x_i) + 2 \sum_{i=2,4,6\ldots}^{m-2} f(x_i) +
			f(x_m) \right]
	\]
	Theoretical error analysis states that the absolute error for Simpson's rule
	is bounded by:
	\[
		E \leq \frac{(b-a)}{180} h^4 \max_{x \in [a,b]} |f^{(4)}(x)| =
		\mathcal{O}(h^4)
	\]
	This implies a theoretical fourth-order convergence ($\mathcal{O}(m^{-4})$),
	provided that the integrand is at least four times continuously
	differentiable, $f \in C^4[a,b]$. The integrals were evaluated using
	double-precision arithmetic (\texttt{Float64}) up to $m = 10,240$ nodes.

	Table~\ref{tab:simpson_results} details the absolute errors obtained at the
	maximum node count ($m = 10,240$) for both the trapezoidal and Simpson's

	\paragraph{Discussion \& Convergence Analysis}
	The integration results perfectly delineate the strict mathematical
	requirements of Simpson's rule:

	\begin{itemize}
		\item \textbf{Smooth Integrands (Integrals 1--4):} These functions satisfy
		      $f \in C^4[a,b]$. Because Simpson's rule scales at $\mathcal{O}(h^4)$, the
		      truncation error diminishes so rapidly that it essentially hits the
		      \texttt{Float64} machine epsilon limit ($\approx 2.22 \times 10^{-16}$) at a
		      fraction of the maximum nodes. Further increases in node count merely
		      accumulate numerical round-off noise.
		\item \textbf{Singular Integrands (Integrals 5--6):} The fourth derivative
		      $f^{(4)}(x)$ approaches infinity at the boundaries for both
		      $\sqrt{x}\log(x)$ and $\sqrt{1-x^2}$. Consequently, the $\mathcal{O}(h^4)$
		      theoretical bound is entirely invalidated. Simpson's rule offers negligible
		      practical improvement over the trapezoidal rule for these functions,
		      remaining trapped at fractional convergence rates due to the dominant root
		      singularities.
	\end{itemize}

\end{solution}

\subsection{Free-nodes Integration}

\begin{exercise}
	Write a program that computes the nodes and weights for the Gauss-Legendre
	quadrature in the interval $[-1,1]$. Below are some tips for the
	implementation.

	\begin{itemize}
		\item To find the roots $x_k$, with $k=1,\ldots,n$ of the Legendre
		      polynomial $P_n(x)$ of order $n$ use Newton's method. Take as the initial
		      condition for the $k$th root,
		      \[
			      x^{(0)}_k=\cos(\phi_k)\,,
			      \qquad
			      \phi_k={4k-1\over 4n+2}\pi\,.
		      \]
		\item In order to use Newton's method you need to evaluate $P_n(x_k^{(i)})$
		      and the derivative $P'_n(x_k^{(i)})$, where $x_k^{(i)}$ is the $i$th
		      iteration in the search for the $k$th root $x_k$. You can obtain these by
		      first solving the recursion relation,

		      \[
			      (m+1)P_{m+1}(x) = (2m+1)xP_{m}(x) - mP_{m-1}(x)\,,
		      \]

		      for $m=1,\ldots,n-1$, where $P_0(x) = 1$ and $P_1(x) = x$, with
		      $x=x^{(i)}_k$. The derivative is then computed using the relation,

		      \[
			      (x^{2}-1)P'_{n}(x)=n[xP_{n}(x)-P_{n-1}(x)]\,.
		      \]

		\item Once you reach the root $x_k$ within some tolerance, the value of the
		      derivative $P'_n(x_k)$ allows you to obtain the corresponding weight using

		      \[
			      w_k= {2\over (1-x_k^2)[P'_n(x_k)]^2}\,.
		      \]
	\end{itemize}

\end{exercise}

\begin{solution}
	The Gauss-Legendre quadrature optimally approximates the integral of a
	function over $[-1, 1]$ using a specific set of points (nodes) and
	corresponding weights. The nodes $x_k$ correspond to the roots of the $n$-th
	degree Legendre polynomial, $P_n(x) = 0$.

	The initial guesses $x_k^{(0)}$ are distributed according to the roots of the
	Chebyshev polynomials, which cluster near the boundaries $[-1, 1]$ and provide
	an excellent starting approximation for Legendre roots. The iteration
	terminates when successive approximations converge within machine precision
	($\approx 10^{-15}$).

	\paragraph{Results \& Validation}
	The algorithm was executed for quadrature orders $n \in \{2, 3, 4, 5\}$. To
	ensure numerical stability and precision, the computed sets of $(x_k, w_k)$
	were benchmarked against the standard \texttt{GaussQuadrature.jl} library. The
	custom implementation successfully matched the verified library routines to an
	absolute tolerance of $\mathcal{O}(10^{-14})$.

	rules with $n=4,6,8,\ldots,40$. Plot the error $|I_n-I|$ as a function of $n$ on
